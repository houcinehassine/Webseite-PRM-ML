🔀 Cross-Validation – Code
# ── K-Fold (Standard) ─────────────────────
scores = cross_val_score(model, X, y, cv=5)
# → array([0.9667, 0.9667, 0.9333, 0.9333, 1.0])
print(scores.mean())  # → 0.96

# ── Leave-One-Out ──────────────────────────
loo = LeaveOneOut()
scores = cross_val_score(
    model, X, y, cv=loo)
# → 150 Werte, mean ≈ 0.96

# ── Two-Fold (manuell) ─────────────────────
X1, X2, y1, y2 = train_test_split(
    X, y, train_size=0.5)
y2_model = model.fit(X1,y1).predict(X2)
y1_model = model.fit(X2,y2).predict(X1)
acc = (accuracy_score(y2,y2_model) +
       accuracy_score(y1,y1_model)) / 2

# ── Repeated K-Fold ────────────────────────
rkf = RepeatedKFold(
    n_splits=5, n_repeats=3, random_state=42)
scores = cross_val_score(model, X, y, cv=rkf)
# → 15 Scores

📈 validation_curve – Vollständiger Code
from sklearn.pipeline import make_pipeline
from sklearn.preprocessing import PolynomialFeatures
from sklearn.linear_model import LinearRegression
from sklearn.model_selection import validation_curve
import numpy as np, matplotlib.pyplot as plt

# Helper
def PolynomialRegression(degree=2, **kw):
    return make_pipeline(
        PolynomialFeatures(degree),
        LinearRegression(**kw))

# validation_curve
degree = np.arange(0, 21)
train_s, val_s = validation_curve(
    PolynomialRegression(), X, y,
    param_name='polynomialfeatures__degree',
    param_range=degree, cv=7)

plt.plot(degree, np.median(train_s,1),
         color='blue', label='training score')
plt.plot(degree, np.median(val_s,1),
         color='red',  label='validation score')
plt.legend(loc='lower center')
plt.ylim(0,1); plt.xlabel('degree')
plt.show()
# → Optimum bei degree ≈ 3-4
📉 learning_curve – Vollständiger Code
from sklearn.model_selection import learning_curve

# Vertikal: (2,1) statt (1,2)!
fig, axes = plt.subplots(2, 1, figsize=(6,8))

for i, degree in enumerate([2, 9]):
    N, train_lc, val_lc = learning_curve(
        PolynomialRegression(degree),
        X, y,
        cv=7,
        train_sizes=np.linspace(0.3, 1, 25))

    axes[i].plot(N, np.mean(train_lc,1),
                 color='blue',
                 label='training score')
    axes[i].plot(N, np.mean(val_lc,1),
                 color='red',
                 label='validation score')
    axes[i].set_ylim(0, 1)
    axes[i].set_title(f'degree = {degree}')
    axes[i].legend(loc='lower right')

plt.tight_layout(); plt.show()
# degree=2: Konvergenz → gut
# degree=9: Gap bleibt → Overfitting
🔧 GridSearchCV – Vollständiger Code
from sklearn.model_selection import GridSearchCV

# 1) param_grid definieren
param_grid = {
    'polynomialfeatures__degree':
        np.arange(21),
    'linearregression__fit_intercept':
        [True, False],
}

# 2) GridSearchCV instanziieren
grid = GridSearchCV(
    PolynomialRegression(),
    param_grid,
    cv=7)

# 3) Fit (alle Kombinationen!)
grid.fit(X, y)

# 4) Ergebnisse ausgeben
print(grid.best_params_)
# {'polynomialfeatures__degree': 4,
#  'linearregression__fit_intercept': False}
print(grid.best_score_)

# 5) Best Model nutzen
model = grid.best_estimator_
y_pred = model.predict(X_new)

# Alternativ: vollständige Pipeline
from sklearn.pipeline import Pipeline
grid2 = GridSearchCV(
    estimator=Pipeline([
        ('polynomialfeatures', PolynomialFeatures()),
        ('linearregression',   LinearRegression())
    ]),
    param_grid=param_grid, cv=7)
🎯 Entscheidungsleitfaden – Was tun wenn...?
❌ accuracy_score = 1.0 auf Trainingsdaten FAIL! Training und Validation auf denselben Daten! → train_test_split verwenden (Seite 6)
⚠️ Training-Score ≈ Validation-Score (beide schlecht) High-Bias / Underfitting → Modellkomplexität erhöhen (z.B. höherer Polynomgrad) oder bessere Features
⚠️ Training-Score hoch, Validation-Score viel niedriger High-Variance / Overfitting → Komplexität reduzieren ODER mehr Trainingsdaten (learning_curve prüfen)
🤔 Welcher Hyperparameter ist optimal? → validation_curve() verwenden · X-Achse = Hyperparameter · Optimum = Maximum des Validation-Scores
🤔 Hilft mehr Daten bei diesem Modell? → learning_curve() verwenden · Wenn Kurven konvergieren bereits = mehr Daten bringen KEINEN Mehrwert mehr
🤔 Welche Hyperparameter-Kombination ist optimal? → GridSearchCV · param_grid definieren · .fit() · .best_params_ lesen
🤔 1 Holdout reicht mir nicht (zu wenig Daten) → cross_val_score(model, X, y, cv=5) · kein Datenverlust · mittlere Performance
🤔 CV-Ergebnisse variieren stark zwischen Runs → RepeatedKFold verwenden · Artefakte durch zufällige Aufteilung vermeiden

AUFGABE TYP 1: Modell trainieren & evaluieren
from sklearn.datasets import load_iris
from sklearn.neighbors import KNeighborsClassifier
from sklearn.model_selection import train_test_split

iris = load_iris()
X, y = iris.data, iris.target

# RICHTIG: Holdout
X_tr, X_te, y_tr, y_te = train_test_split(
    X, y, train_size=0.5, random_state=42)

model = KNeighborsClassifier(n_neighbors=1)
model.fit(X_tr, y_tr)
print(model.score(X_te, y_te))  # 0.96
AUFGABE TYP 2: Cross-Validation
from sklearn.model_selection import cross_val_score
from sklearn.neighbors import KNeighborsClassifier

model = KNeighborsClassifier(n_neighbors=1)

# Standard K-Fold
scores = cross_val_score(
    model, X, y, cv=5)
print(scores.mean().round(4))  # 0.96

# Interpretation:
# Mittelwert = globale Performance
# Std = Stabilität des Modells
